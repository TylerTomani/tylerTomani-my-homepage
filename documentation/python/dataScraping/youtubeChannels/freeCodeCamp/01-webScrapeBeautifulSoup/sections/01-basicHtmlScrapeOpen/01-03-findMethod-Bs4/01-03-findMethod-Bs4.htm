Section 01-02 - find() and find_all() - BeautifulSoup

01

find()

find() will get the first instance of the searched tag 

from bs4 import BeautifulSoup

with open('home.htm', 'r') as f:
content = f.read()
soup = BeautifulSoup(content,'html.parser')
tags = soup.find('h5')

02

find_all()

Use find all to get every instance of the element 

from bs4 import BeautifulSoup
with open('home.htm','r') as f:
content = f.read()
soup = BeautifulSoup(content,'html.parser')
tags = soup.find_all('h5')

03

print each <h5> tag containing the course name 

from bs4 import BeautifulSoup
with open('home.htm','r') as f:
    content = f.read()
    soup = BeautifulSoup(content,'html.parser')
    courses_html_tags = soup.find_all('h5')
    for course in courses_html_tags:
        print(course)

04

print only the text of each <h5> course name in a for loop 

from bs4 import BeautifulSoup
with open('home.htm','r') as f:
    content = f.read()
    soup = BeautifulSoup(content,'html.parser')
    courses_html_tags = soup.find_all('h5')
    for course in courses_html_tags:
        print(course.text)
