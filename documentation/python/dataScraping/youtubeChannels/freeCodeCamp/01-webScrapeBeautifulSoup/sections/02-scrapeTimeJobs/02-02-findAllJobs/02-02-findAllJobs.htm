from bs4 import BeautifulSoup
import requests
html =
requests.get('https://www.timesjobs.com/candidate/job-search.html?searchType=personalizedSearch&from=submit&searchTextSrc=&searchTextText=&txtKeywords=python&txtLocation=')
html_text = html.text
soup = BeautifulSoup(html_text,'html.parser')
jobs = soup.find_all('li',class_='clearfix job-bx wht-shd-bx')
for job in jobs:
    company_name = job.find('h3',class_='joblist-comp-name').text
    skills = job.find('span',class_="srp-skills").text.replace(' ','')
    data_published = job.find('span',class_='sim-posted').span.text
    print(company_name)
    print(skills)
    print(data_published)
    break
