def get_clinic_name(clinic_id):
    url = f'https://{clinic_id}.portal.athenahealth.com/'
    response = requests.get(url)
    html = response.text
    soup = BeautifulSoup(html,'html.parser')
    clinic_name = soup.find_all('h1').text.strip()
    return clinic_name
get_clinic_name(12693)

import requests
from bs4 import BeautifulSoup

def get_clinic_name(clinic_id):
    url = f'https://{clinic_id}.portal.athenahealth.com/'
    response = requests.get(url)
    html = response.text
    soup = BeautifulSoup(html,'html.parser')
    clinic_name = soup.find_all('h1')[-1].text.strip()
    return clinic_name


main_list = []
start = 12690
end = 12700
for clinic_id in range(start,end+1):
    data_dict = {}
    data_dict['clinic_id'] = clinic_id
    data_dict['name'] = get_clinic_name(clinic_id)
    if data_dict['name'] != 'Payment Confirmation' and data_dict['name'] != "Sorry, we can't find that practice. Make sure you typed the right address.":
        main_list.append(data_dict)

import pandas as pd
df = pd.DataFrame(main_list)
df.to_csv('clinic_data.csv',index=False)
