def lowercase(txt):
    return txt.lower()
def removePunctuation(txt):
    punctuations = ['.', '-', ',', '*']
    for punctuation in punctuations:
        txt = txt.replace(punctuations,'')
    return txt
def removeNewLines(txt):
    txt = txt.replace('\n','')
return txt
