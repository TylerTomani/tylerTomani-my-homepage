def factorial(n):
    if type(n) is not int:
        return None
    if n < 0:
        return None
    if n == 0: 
        return 1

    i = 0
    f = 1
    while i < n:
        i =  i + 1
        f = f * i
    return f

factorial(4)

Recursion

def factorial(num):
    if type(num) is not int:
        return None
    if num < 0:
        return None
    if num == 0:
        return 1

    return num * factorial(num - 1)
print(factorial(5))
