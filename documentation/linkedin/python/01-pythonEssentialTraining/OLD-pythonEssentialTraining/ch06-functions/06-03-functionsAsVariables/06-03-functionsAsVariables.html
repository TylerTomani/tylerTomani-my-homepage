Below are different functions that we can place into a dictonaries 

def lowercase(text):
    return text.lower()
def removePunctuation(text):
    punctuations = ['.', '-', ',', '*']
    for punctuation in punctuations:
        text = text.replace(punctuations,'')
    return text
def removeNewLines(text):
    text = text.replace('\n','')
    return text
